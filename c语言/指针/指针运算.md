# 1指针加减整数
```c
#define n 5
float values[n];
float *vp;

//指针加减整数
for (vp = &values[0], vp < values[n]; )//无需步骤直接继续循环
{
	*vp++ = 0;//vp = 0; vp++;(后置++)
	
	}//程序相当于给values【0】->values[n]都赋值为0
 *vp++为给地址++
 （*vp）++为给对象++
```

```c
int main(){
	
	//f1正常赋值
	int arr[10];
	int i = 0;
	int sz = sizeof(arr) / sizeof(arr[0]);
	//计算数组单元数数组的总大小 / 数组中每个元素的大小 = 元素个数
	for(int i = 0; i < sz; i++)
	arr[i] = 1;
	
	
	//f2指针法
	int* p = arr;
	for (i = 0; i < sz; i++){\
		*p = 1;
		p++; //或者直接*(p+i) = 1
		}
		
	
	
}
```
